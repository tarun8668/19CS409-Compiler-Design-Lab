	%{ 
	/* Definition section */

	/* Rule Section */
	%% 
	[0-9]+ { 
		yylval=atoi(yytext); 
		return NUMBER; 

	} 
	'<=' return LE;
	'>=' return GE;
	'!=' return NE;
	'==' return EQ;
	[\t] ; 
	[\n] return 0;

	%%
